

#include <stdio.h>
 
int main() {
 
    int n;
    double x[12][12], soma = 0, media = 0;
    char t;

    scanf("%d %c", &n, &t);

    for(int i=0; i<12; i++){
        for(int j=0; j<12; j++){
            scanf("%lf", &x[i][j]);
        }
    } 

    for(int i=0; i<12; i++){
        if(t == 'S') soma += x[i][n];
        else if(t == 'M') media += x[i][n];
    }

    if(t == 'S') printf("%.1lf\n", soma);
    else if(t == 'M') printf("%.1lf\n", media / 12);
   
    return 0;
}
